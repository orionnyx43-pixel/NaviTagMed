        // 1. Kunin ang encoded data mula sa URL (?d=...)
        const urlParams = new URLSearchParams(window.location.search);
        const encodedData = urlParams.get('d');

        if (encodedData) {
            try {
                // 2. I-decode ang Base64 JSON data
                const d = JSON.parse(decodeURIComponent(escape(atob(encodedData))));

                // 3. I-display sa document format
                document.getElementById('docContent').innerHTML = `
                    <div class="section-title">Patient Identity</div>
                    <div class="info-row">
                        <div class="label">Full Name</div>
                        <div class="value">${d.name}</div>
                    </div>
                    <div style="display:flex; justify-content: space-between;">
                        <div class="info-row">
                            <div class="label">Age</div>
                            <div class="value">${d.age} yrs old</div>
                        </div>
                        <div class="info-row" style="text-align:right;">
                            <div class="label">Blood Type</div>
                            <div class="blood-badge">${d.blood}</div>
                        </div>
                    </div>

                    <div class="section-title">Critical Medical Info</div>
                    <div class="critical-box">
                        <div class="label" style="color:#C53030;">⚠️ Allergies</div>
                        <div class="value">${d.allergies}</div>
                    </div>
                    <div class="info-row">
                        <div class="label">Medical History / Background</div>
                        <div class="value">${d.bg}</div>
                    </div>
                    <div class="info-row">
                        <div class="label">Maintenance Medications</div>
                        <div class="value">${d.meds}</div>
                    </div>

                    <div class="section-title">Emergency Contact</div>
                    <div class="info-row">
                        <div class="label">Primary Relative</div>
                        <div class="value">${d.relName}</div>
                        <div class="value" style="color:var(--red);">${d.relNum}</div>
                    </div>

                    <div class="section-title">Notes for First Aiders</div>
                    <div class="info-row" style="background:#f1f3f5; padding:10px; border-radius:8px; font-style:italic;">
                        "${d.notes}"
                    </div>
                `;
                document.getElementById('serial').innerText = d.s || "N/A";

            } catch (e) {
                document.getElementById('docContent').innerHTML = "<p style='color:red;'>Error: Data corruption or invalid QR code.</p>";
            }
        } else {
            document.getElementById('docContent').innerHTML = "<p>No medical data found. Please scan a valid Navi Tag.</p>";
        }
    </script>
